// 🎮 गेमचे वेरिएबल्स let score = 0; let totalQuestions = 0; let correctAnswer = 0; let audioCtx = null; // ब्राउझरसाठी ऑडिओ कॉन्टेक्स्ट // 🔊 ऑडिओ सिस्टीम सुरू करणे (GitHub/Browser सुरक्षेसाठी आवश्यक) function initAudio() { if (!audioCtx) { audioCtx = new (window.AudioContext || window.webkitAudioContext)(); } if (audioCtx.state === 'suspended') { audioCtx.resume(); } } // युझरने स्क्रीनवर कुठेही पहिली क्लिक किंवा टच करताच ऑडिओ परवानगी ॲक्टिव्हेट होईल window.addEventListener('click', initAudio, { once: true }); window.addEventListener('touchstart', initAudio, { once: true }); // 🔔 नवीन प्रश्न तयार करणे function generateQuestion() { // १ ते २० मधील कोणतेही दोन अंक let num1 = Math.floor(Math.random() * 20) + 1; let num2 = Math.floor(Math.random() * 20) + 1; // बेरीज किंवा वजाबाकी पैकी एक निवडणे let operation = Math.random() > 0.5 ? '+' : '-'; // वजाबाकी असेल तर उत्तर उणे (Minus) येऊ नये म्हणून मोठा अंक आधी ठेवू if (operation === '-') { if (num1 < num2) { let temp = num1; num1 = num2; num2 = temp; } correctAnswer = num1 - num2; } else { correctAnswer = num1 + num2; } // स्क्रीनवर प्रश्न दाखवणे document.getElementById('question').innerText = `${num1} ${operation} ${num2} = ?`; document.getElementById('user-answer').value = ''; document.getElementById('user-answer').focus(); } // 🔊 बरोबर उत्तरासाठी गोड ट्यून (Synthesized Sound) function playCorrectSound() { try { initAudio(); if (!audioCtx) return; let osc = audioCtx.createOscillator(); let gain = audioCtx.createGain(); osc.type = 'sine'; osc.connect(gain); gain.connect(audioCtx.destination); // पहिली टीप (D5) osc.frequency.setValueAtTime(587.33, audioCtx.currentTime); gain.gain.setValueAtTime(0.1, audioCtx.currentTime); osc.start(); // दुसरी उच्च टीप (A5) osc.frequency.setValueAtTime(880, audioCtx.currentTime + 0.1); gain.gain.setValueAtTime(0.1, audioCtx.currentTime + 0.1); gain.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime + 0.4); osc.stop(audioCtx.currentTime + 0.4); } catch (e) { console.log("Audio play blocked by browser:", e); } } // 🔊 चुकीच्या उत्तरासाठी बझर आवाज function playWrongSound() { try { initAudio(); if (!audioCtx) return; let osc = audioCtx.createOscillator(); let gain = audioCtx.createGain(); osc.type = 'sawtooth'; osc.connect(gain); gain.connect(audioCtx.destination); osc.frequency.setValueAtTime(130.81, audioCtx.currentTime); gain.gain.setValueAtTime(0.15, audioCtx.currentTime); osc.start(); gain.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime + 0.3); osc.stop(audioCtx.currentTime + 0.3); } catch (e) { console.log("Audio play blocked by browser:", e); } } // ✅ उत्तर तपासणे function checkAnswer() { let userAnswer = parseInt(document.getElementById('user-answer').value); let resultDiv = document.getElementById('result'); // जर इनपुट रिकामे असेल तर if (isNaN(userAnswer)) { resultDiv.className = "result-message wrong"; resultDiv.innerText = "कृपया आधी उत्तर टाईप करा!"; return; } totalQuestions++; document.getElementById('total-questions').innerText = totalQuestions; if (userAnswer === correctAnswer) { score++; document.getElementById('score').innerText = score; resultDiv.className = "result-message correct"; resultDiv.innerText = "शाब्बास! बरोबर उत्तर आहे 🎉"; playCorrectSound(); // 🔊 बरोबर आवाज } else { resultDiv.className = "result-message wrong"; resultDiv.innerText = `चूक! योग्य उत्तर ${correctAnswer} होते. ❌`; playWrongSound(); // 🔊 चुकीचा आवाज } // २ सेकंदानंतर नवीन प्रश्न समोर येईल setTimeout(generateQuestion, 2000); } // ⌨️ कीबोर्डवरील 'Enter' बटण दाबूनही उत्तर तपासता येईल document.getElementById('user-answer').addEventListener('keyup', function(event) { if (event.key === 'Enter') { checkAnswer(); } }); // खेळ सुरू झाल्यावर पहिला प्रश्न लोड करणे window.onload = generateQuestion;
